    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null) {
            return l2;
        }
        if (l2 == null) {
            return l1;
        }

        ListNode root;
        if ( l1.val >= l2.val) {
            root = new ListNode(l2.val);
            l2 = l2.next;
        } else {
            root = new ListNode(l1.val);
            l1 = l1.next;
        }
        ListNode node = root;
        while (l1 != null && l2 != null) {
            if (l1.val >= l2.val) {
                node.next = new ListNode(l2.val);
                l2 = l2.next;
                node = node.next;
            } 
            if (l2 != null && l2.val >= l1.val) {
                node.next = new ListNode(l1.val);
                l1 = l1.next;
                node = node.next;
            }
        }

        if (l1 == null) {
            while (l2 != null) {
                node.next = new ListNode(l2.val);
                l2 = l2.next;
                node = node.next;
            }
        }
        if (l2 == null) {
            while (l1 != null) {
                node.next = new ListNode(l1.val);
                l1 = l1.next;
                node = node.next;
            }
        }
        return root;
    }
