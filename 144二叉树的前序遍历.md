List<Integer> resList = new ArrayList<>();
public List<Integer> preorderTraversal(TreeNode root) {
    if (root == null) {
        return resList;
    }
    preOrder(root);
    return resList;
}
public void preOrder(TreeNode node) {
    if (node == null) {
        return;
    }
    resList.add(node.val);
    preOrder(node.left);
    preOrder(node.right);
}
