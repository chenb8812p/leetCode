public int[] advantageCount(int[] A, int[] B) {
        Arrays.sort(A);
        List<Integer> aList = new ArrayList<>();
        for (Integer i : A) {
            aList.add(i);
        }

        for (int i = 0; i < B.length; i++) {
            int bInt = B[i];

            for (int i1 = 0; i1 < aList.size(); i1++) {
                Integer aInt = aList.get(i1);
                if (aInt > bInt) {
                    A[i] = aInt;
                    aList.remove(i1);
                    break;
                } else {
                    A[i] = -1;
                }
            }
        }

        for (int i = 0; i < A.length; i++) {
            if (A[i] == -1) {
                A[i] = aList.get(0);
                aList.remove(0);
            }
        }
        return A;
    }
